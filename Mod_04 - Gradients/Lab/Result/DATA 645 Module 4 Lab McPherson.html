# MSE Cost Function

def mse_cost_function(theta):
    """
    theta is a length-2 vector [theta0, theta1]
    MSE = (1/m) * sum((X_b @ theta - y)^2)
    """
    theta = theta.reshape(2, 1)           # ensure shape (2,1)
    errors = X_b @ theta - y              # shape (100,1)
    return float((errors.T @ errors) / len(y))  # scalar

def gradient_descent(initial_guess, learning_rate, num_iterations):
    trajectory = []   # Store theta values for plotting
    cost = []        # Store MSE for each iteration

    current_point = np.array(initial_guess, dtype=float).reshape(2, 1)  

    for iteration in range(num_iterations):
        trajectory.append(current_point.copy())
        cost.append(mse_cost_function(current_point))  

        # Using MSE
        m = len(y)
        gradient = (2/m) * (X_b.T @ (X_b @ current_point - y))

        # same update rule form: point = point - lr * gradient
        current_point = current_point - learning_rate * gradient

    return np.array(trajectory), np.array(cost), current_point

# MSE Cost Function

def mse_cost_function(theta):
    """
    theta is a length-2 vector [theta0, theta1]
    MSE = (1/m) * sum((X_b @ theta - y)^2)
    """
    theta = theta.reshape(2, 1)           # ensure shape (2,1)
    errors = X_b @ theta - y              # shape (100,1)
    return float((errors.T @ errors) / len(y))  # scalar

def sto_gradient_descent(initial_guess, learning_rate, num_iterations):
    trajectory = []   # Store theta values for plotting
    cost = []        # Store MSE for each iteration

    current_point = np.array(initial_guess, dtype=float).reshape(2, 1)

    # (CHANGE 1) Precompute m once (not required, just avoids repeating len(y))
    m = len(y)

    for iteration in range(num_iterations):
        trajectory.append(current_point.copy())
        cost.append(mse_cost_function(current_point))

        # (CHANGE 2) Choose ONE random training example index i
        i = np.random.randint(m)

        # (CHANGE 3) Grab that single example (Xi, yi)
        Xi = X_b[i:i+1, :]   # shape (1,2)
        yi = y[i:i+1, :]     # shape (1,1)

        # (CHANGE 4) Stochastic gradient using only that one example:
        # grad = 2 * Xi^T (Xi*theta - yi)
        gradient = 2 * (Xi.T @ (Xi @ current_point - yi))

        # same update rule form: point = point - lr * gradient
        current_point = current_point - learning_rate * gradient

    return np.array(trajectory), np.array(cost), current_point

def mini_gradient_descent(initial_guess, learning_rate, num_iterations, batch_size=10):
    trajectory = []   # Store theta values for plotting
    cost = []        # Store MSE for each iteration

    current_point = np.array(initial_guess, dtype=float).reshape(2, 1)

    m = len(y)

    for iteration in range(num_iterations):
        trajectory.append(current_point.copy())
        cost.append(mse_cost_function(current_point))

        # (CHANGE 1) Choose a RANDOM MINI-BATCH of indices (instead of one index)
        batch_idx = np.random.randint(m, size=batch_size)

        # (CHANGE 2) Grab the mini-batch (X_batch, y_batch)
        X_batch = X_b[batch_idx, :]   # shape (batch_size, 2)
        y_batch = y[batch_idx, :]     # shape (batch_size, 1)

        # (CHANGE 3) Mini-batch gradient:
        # grad = (2/b) X_batch^T (X_batch*theta - y_batch)
        b = batch_size
        gradient = (2/b) * (X_batch.T @ (X_batch @ current_point - y_batch))

        # same update rule form: point = point - lr * gradient
        current_point = current_point - learning_rate * gradient

    return np.array(trajectory), np.array(cost), current_point

4.a Compare stochastic gradient descent, gradient descent, and mini-batch gradient descent in terms of the paths taken by the three gradient descent algorithms in the parameter space during training. What do you observe? Which algorithm shows faster convergence?

Batch Gradient descent appears to have the smoothest and fastest path towards the convergence on an iteration-by-iteration basis, enabling it to use higher learning rates between iterations. However, this is presumably at a cost of computational cost, with each iteration being more expensive. Stochastic Gradient Descent appears to need a lower learning rate to be effective, since its random approach, combined with the small sample size, creates the possibility of overshooting convergence. Its path is less smooth that Batch. Mini-batch approaches convergence at almost the same rate that Batch approaches convergence, but presumably does it "cheaper" on an iteration by iteration basis. Its path is smoother than SGD but less smooth than batch. Overall, mini-batch appears to provide similar perfromance at a presumably lower cost.
